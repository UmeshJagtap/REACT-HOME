//  Single Cat Image

import React, { useState } from 'react';

export default function App() {
  const [data, setData] = useState([
    {
      id: 'gCEFbK7in',
      url: 'https://cdn2.thecatapi.com/images/gCEFbK7in.jpg',
      width: 1424,
      height: 987,
    },
  ]);

  const fetchData1 = () => {
    const url = 'https://api.thecatapi.com/v1/images/search?breed_id=abys';
    // const url = 'https://api.thecatapi.com/v1/breeds';
    fetch(url)
      .then((response) => response.json())
      .then((response) => setData(response));
    // .then((json) => console.log(json));
  };

  return (
    <div>
      <h1>Cat API</h1>
      <button onClick={fetchData1}>Fetch New Cat</button>
      <br />
      {JSON.stringify(data)}
      {data.map((cat) => {
        return (
          <div key={cat.id}>
            <img
              src={cat.url}
              alt="catPic"
              style={{ height: 500, width: 500 }}
            />
          </div>
        );
      })}
    </div>
  );
}

import React, { useState } from 'react';

export default function App() {
  const [data, setData] = useState([]);

  const fetchData1 = () => {
    // const url = 'https://api.thecatapi.com/v1/images/search?breed_id=abys';
    const url = 'https://api.thecatapi.com/v1/breeds';
    fetch(url)
      .then((response) => response.json())
      .then((response) => setData(response));
    // .then((json) => console.log(json));
  };

  return (
    <div>
      <h1>Cat API</h1>
      <button onClick={fetchData1}>Fetch New Cat</button>
      <br />
      {/* {JSON.stringify(data)} */}
      {data.map((cat) => {
        return (
          <div key={cat.id}>
            <div className="">
              <h3>{cat.name}</h3>
              <h4>Origin : {cat.origin}</h4>
              <h4>Lifespan : {cat.life_span} years.</h4>
              <p>{cat.description}</p>
              <p>{cat.temperament}</p>
              <img
                src={`https://cdn2.thecatapi.com/images/${cat.reference_image_id}.jpg`}
                alt="catPic"
                style={{ height: 500, width: 500 }}
              />
              <hr />
            </div>
          </div>
        );
      })}
    </div>
  );
